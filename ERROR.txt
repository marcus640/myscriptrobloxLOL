local LP = game:GetService("Players").LocalPlayer
local UIS = game:GetService("UserInputService")
local UI_NAME = "H2K4R_V12_FINAL"

pcall(function()
    if LP.PlayerGui:FindFirstChild(UI_NAME) then LP.PlayerGui[UI_NAME]:Destroy() end
end)

local sg = Instance.new("ScreenGui")
sg.Name = UI_NAME
sg.Parent = LP:WaitForChild("PlayerGui")
sg.ResetOnSpawn = false

-- [[ ВІКНО ]] --
local Frame = Instance.new("Frame", sg)
Frame.Size = UDim2.new(0, 320, 0, 400)
Frame.Position = UDim2.new(0.5, -160, 0.5, -200)
Frame.BackgroundColor3 = Color3.fromRGB(15, 0, 0)
Frame.BorderSizePixel = 2
Frame.BorderColor3 = Color3.fromRGB(255, 0, 0)
Frame.Visible = false 
Frame.Active = true

-- ЗАГОЛОВОК
local Title = Instance.new("TextButton", Frame)
Title.Size = UDim2.new(1, 0, 0, 35)
Title.BackgroundColor3 = Color3.fromRGB(80, 0, 0)
Title.Text = "H2K4R HUB V12"
Title.TextColor3 = Color3.new(1, 1, 1)
Title.Font = Enum.Font.Code
Title.TextSize = 16

-- [[ ТАБИ ]] --
local TabFrame = Instance.new("Frame", Frame)
TabFrame.Size = UDim2.new(1, 0, 0, 35)
TabFrame.Position = UDim2.new(0, 0, 0, 35)
TabFrame.BackgroundColor3 = Color3.fromRGB(30, 0, 0)

local function CreateTabBtn(text, pos, width)
    local b = Instance.new("TextButton", TabFrame)
    b.Size = UDim2.new(width, 0, 1, 0)
    b.Position = UDim2.new(pos, 0, 0, 0)
    b.Text = text
    b.BackgroundColor3 = Color3.fromRGB(20, 0, 0)
    b.TextColor3 = Color3.new(1, 1, 1)
    b.Font = Enum.Font.Code
    b.TextSize = 10
    return b
end

local MainTabBtn = CreateTabBtn("MAIN", 0, 0.33)
local FeTabBtn = CreateTabBtn("FE SCRIPTS", 0.33, 0.33)
local HubTabBtn = CreateTabBtn("HUB", 0.66, 0.34)

local function CreateScroll()
    local s = Instance.new("ScrollingFrame", Frame)
    s.Size = UDim2.new(1, -10, 1, -85)
    s.Position = UDim2.new(0, 5, 0, 80)
    s.BackgroundTransparency = 1; s.Visible = false
    s.CanvasSize = UDim2.new(0, 0, 0, 500); s.ScrollBarThickness = 3
    local l = Instance.new("UIListLayout", s); l.Padding = UDim.new(0, 5)
    return s
end

local MainS = CreateScroll(); local FeS = CreateScroll(); local HubS = CreateScroll()
MainS.Visible = true; MainTabBtn.BackgroundColor3 = Color3.fromRGB(80, 0, 0)

local function Add(name, url, parent)
    local b = Instance.new("TextButton", parent)
    b.Size = UDim2.new(1, -5, 0, 35); b.BackgroundColor3 = Color3.fromRGB(30, 0, 0)
    b.BorderSizePixel = 1; b.BorderColor3 = Color3.fromRGB(200, 0, 0)
    b.Text = "> " .. name; b.TextColor3 = Color3.new(1, 1, 1); b.Font = Enum.Font.Code; b.TextSize = 11
    b.TextXAlignment = Enum.TextXAlignment.Left
    b.MouseButton1Down:Connect(function()
        if type(url) == "string" then loadstring(game:HttpGet(url))() else url() end
    end)
end

Add("INF JUMP", function() UIS.JumpRequest:Connect(function() if LP.Character then LP.Character:FindFirstChildOfClass("Humanoid"):ChangeState("Jumping") end end) end, MainS)

-- [[ ЛОГІКА ТАБІВ ]] --
local function ShowTab(tab, btn)
    MainS.Visible = false; FeS.Visible = false; HubS.Visible = false
    MainTabBtn.BackgroundColor3 = Color3.fromRGB(20, 0, 0); FeTabBtn.BackgroundColor3 = Color3.fromRGB(20, 0, 0); HubTabBtn.BackgroundColor3 = Color3.fromRGB(20, 0, 0)
    tab.Visible = true; btn.BackgroundColor3 = Color3.fromRGB(80, 0, 0)
end
MainTabBtn.MouseButton1Down:Connect(function() ShowTab(MainS, MainTabBtn) end)
FeTabBtn.MouseButton1Down:Connect(function() ShowTab(FeS, FeTabBtn) end)
HubTabBtn.MouseButton1Down:Connect(function() ShowTab(HubS, HubTabBtn) end)

-- РУХ ГУІ
local drag, dStart, sPos
Title.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then drag = true dStart = i.Position sPos = Frame.Position end end)
UIS.InputChanged:Connect(function(i) if drag and i.UserInputType == Enum.UserInputType.MouseMovement then local delta = i.Position - dStart
Frame.Position = UDim2.new(sPos.X.Scale, sPos.X.Offset + delta.X, sPos.Y.Scale, sPos.Y.Offset + delta.Y) end end)
UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then drag = false end end)

OpenBtn.BackgroundColor3 = Color3.fromRGB(150, 0, 0); OpenBtn.Text = "[ OPEN ]"; OpenBtn.TextColor3 = Color3.new(1, 1, 1)
OpenBtn.Font = Enum.Font.Code; OpenBtn.BorderSizePixel = 2; OpenBtn.BorderColor3 = Color3.new(0, 0, 0)
OpenBtn.MouseButton1Down:Connect(function() Frame.Visible = not Frame.Visible; OpenBtn.Text = Frame.Visible and "[ CLOSE ]" or "[ OPEN ]" end)
