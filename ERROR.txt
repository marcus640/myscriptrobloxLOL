    btn.Size = UDim2.new(0.9, 0, 0, 40)
    btn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)

-- 1. TP Tool (–Ü–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç –¥–ª—è —Ç–µ–ª–µ–ø–æ—Ä—Ç—É)
CreateBtn("üìç Get TP Tool", function()
    local mouse = LP:GetMouse()
    local tool = Instance.new("Tool")
    tool.RequiresHandle = false
    tool.Name = "TP Tool"
    tool.Activated:Connect(function()
        local pos = mouse.Hit.p
        LP.Character:MoveTo(pos + Vector3.new(0, 3, 0))
    end)
    tool.Parent = LP.Backpack
end)

-- 2. ESP (–ë–∞—á–∏—Ç–∏ –≥—Ä–∞–≤—Ü—ñ–≤)
local espEnabled = false
CreateBtn("üëÅÔ∏è ESP (Wallhack)", function()
    espEnabled = not espEnabled
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            local highlight = p.Character:FindFirstChild("DarkESP")
            if espEnabled then
                if not highlight then
                    highlight = Instance.new("Highlight")
                    highlight.Name = "DarkESP"
                    highlight.FillColor = Color3.fromRGB(255, 0, 0)
                    highlight.Parent = p.Character
                end
            else
                if highlight then highlight:Destroy() end
            end
        end
    end
end)
