local function AddButton(text, yPos, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0.9, 0, 0, 45)
    btn.Position = UDim2.new(0.05, 0, 0, yPos)
    btn.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    btn.Text = text
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.GothamSemibold
    btn.TextSize = 14
    btn.Parent = MainFrame
    
    local corner = Instance.new("UICorner")
    corner.Parent = btn
    
    btn.MouseButton1Click:Connect(callback)
end

-- 4. АНІМАЦІЯ ВІДКРИТТЯ
local isOpen = false
LogoButton.MouseButton1Click:Connect(function()
    if not isOpen then
        MainFrame:TweenPosition(UDim2.new(0.5, -150, 0.5, -175), "Out", "Back", 0.5, true)
    else
        MainFrame:TweenPosition(UDim2.new(0.5, -150, 1.2, 0), "In", "Back", 0.5, true)
    end
    isOpen = not isOpen
end)

-- 5. DRAG (Перетягування кнопки)
local UserInputService = game:GetService("UserInputService")
local dragging, dragInput, dragStart, startPos

LogoButton.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = LogoButton.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        LogoButton.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

LogoButton.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)
