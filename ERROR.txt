-- [[ ФУНКЦІЯ НЕОНОВОГО КONTУРУ (ЛІНІЯ, ЩО РУХАЄТЬСЯ) ]] --
local function ApplyNeonBorder(instance)
    local stroke = Instance.new("UIStroke", instance)
    stroke.Thickness = 2
    stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    
    local gradient = Instance.new("UIGradient", stroke)
    gradient.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, Color3.fromRGB(150, 0, 255)),   -- Фіолетовий
        ColorSequenceKeypoint.new(0.25, Color3.fromRGB(255, 0, 150)), -- Маджента (неон)
        ColorSequenceKeypoint.new(0.5, Color3.fromRGB(20, 0, 40)),    -- Темна зона для контрасту
        ColorSequenceKeypoint.new(0.75, Color3.fromRGB(255, 0, 150)), -- Маджента (неон)
        ColorSequenceKeypoint.new(1, Color3.fromRGB(150, 0, 255))    -- Фіолетовий
    })
    
    -- Плавне обертання лінії в реальному часі
    RS.RenderStepped:Connect(function(dt)
        if gradient and gradient.Parent then
            gradient.Rotation = (gradient.Rotation + 140 * dt) % 360
        end
    end)
end

-- [[ ФУНКЦІЯ ПЕРЕЛИВАННЯ БЛИСКОМ (SHIMMER EFFECT) ]] --
local function AddShimmerEffect(btn)
    btn.ClipsDescendants = true -- Щоб блиск не виходив за межі кнопки
    
    local shine = Instance.new("Frame", btn)
    shine.Size = UDim2.new(0, 50, 2, 0)
    shine.Position = UDim2.new(-2, 0, -0.5, 0)
    shine.Rotation = 30
    shine.BackgroundColor3 = Color3.new(1, 1, 1)
    shine.BorderSizePixel = 0
    
    local gradient = Instance.new("UIGradient", shine)
    gradient.Color = ColorSequence.new(Color3.new(1, 1, 1))
    gradient.Transparency = NumberSequence.new({
        NumberSequenceKeypoint.new(0, 1),
        NumberSequenceKeypoint.new(0.5, 0.4), -- Максимальна яскравість по центру лінзи
        NumberSequenceKeypoint.new(1, 1)
    })
    
    -- Циклічний запуск анімації блиску
    task.spawn(function()
        while btn and btn.Parent do
            task.wait(math.random(3, 5)) -- Випадкова пауза між переливами
            shine.Position = UDim2.new(-1.5, 0, -0.5, 0)
            local tween = TS:Create(shine, TweenInfo.new(1.2, Enum.EasingStyle.Linear), {Position = UDim2.new(2, 0, -0.5, 0)})
            tween:Play()
            tween.Completed:Wait()
        end
    end)
end

    local fillC = Instance.new("UICorner", fill); fillC.CornerRadius = UDim.new(1, 0)

        local move; move = UIS.InputChanged:Connect(function(input)

                local rel = math.clamp((input.Position.X - bar.AbsolutePosition.X) / bar.AbsoluteSize.X, 0, 1)

            end
        end)

    end)
end

-- [[ DRAG & CLICK ]] --
local function MakeDraggable(obj, target)
    local drag, start, sPos
    obj.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then drag = true start = i.Position sPos = target.Position end end)
    UIS.InputChanged:Connect(function(i) if drag and i.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = i.Position - start
        target.Position = UDim2.new(sPos.X.Scale, sPos.X.Offset + delta.X, sPos.Y.Scale, sPos.Y.Offset + delta.Y)
    end end)
    UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then drag = false end end)
end
